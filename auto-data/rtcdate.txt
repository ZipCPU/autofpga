@DEVID=RTCDATE
@NADDR=1
@ACCESS=@$(DEVID)_ACCESS
@DEPENDS=RTC_ACCESS
@SLAVE.TYPE=SINGLE
@SLAVE.BUS=wb
@MAIN.INSERT=
	//
	// The Calendar DATE

@BDEF.OSVAL=	static volatile @$(BDEF.IOTYPE) *const @$(BDEF.IONAME) = ((@$BDEF.IOTYPE *)@$REGBASE);
